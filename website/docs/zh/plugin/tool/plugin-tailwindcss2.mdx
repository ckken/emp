import { Badge } from '@theme';

## 快速开始

### 安装插件
import { PackageManagerTabs } from '@theme';

<PackageManagerTabs command="add @empjs/plugin-tailwindcss2" />

### 注册插件
```ts title="emp-config.js"
import {defineConfig} from '@empjs/cli'
import pluginTailwindcss from '@empjs/plugin-tailwindcss2'
export default defineConfig(store => {
  return {
    plugins:[
      pluginTailwindcss(),
    ],
  }
})

```

### 引入样式
```css title="style.css"
@tailwind base;
@tailwind components;
@tailwind utilities;
```
#### 引入 style.css 到项目
```ts
import 'src/style.css'
```
