## css.sass
> sass build supports two modes: `default` and `modern`

<SassModern />

+ Type
```ts
  api?: 'modern' | 'modern-compiler'
  sassOptions?: SassOptions<'async'>
  mode?: 'default' | 'modern'
  implementation?: object | string
```

+ Default
```js
export default {
  css:{
      sass:{
        mode:'modern'
      }
  }
}
```

```ts title="emp-config.js"
import {defineConfig} from '@empjs/cli'
export default defineConfig(store => {
  return {
    css:{
        // Enable sass modern build
        sass:{
          mode:'modern'
        }
    }
  }
})
```
